Table scrapping with Beautiful Soup¶



page = requests.get(url).text



soup = BeautifulSoup(page)
tables = soup.find_all('table')
course_table = tables[5]
rows =  course_table.find_all("tr")

data = []
for row in rows[4:]:
    d = [s.get_text().strip() for s in row.find_all("td")]
    if "Class" not in d[0]:
        link = "" if row.find("a") is None else row.find("a")['href']
        d.append(link)
        data.append(d)
headers = list(rows[3].stripped_strings)
headers.append("Link")

df = pd.DataFrame(data, columns = headers)

Searching for enrollment information¶



def get_enrollment(url):
    s = BeautifulSoup(requests.get(url).text)
    cap =  [td for td in s.find_all('td') if "Enrollment Capacity:" in td.get_text()][0].find_next('td')
    tot =  [td for td in s.find_all('td') if "Enrollment Total:" in td.get_text()][0].find_next('td')
    cap = int(cap.get_text())
    tot = int(tot.get_text())
    return cap, tot
